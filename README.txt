  M3U8 Source Resolver — تعليمات التشغيل
══════════════════════════════════════

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  التشغيل عبر Docker (موصى به)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. بناء الصورة:
   docker build -t m3u8-resolver .

2. التشغيل مع كلمة مرور:
   docker run -p 3000:3000 -e RESOLVER_PASSWORD=كلمتك_هنا m3u8-resolver

3. افتح المتصفح على:
   http://localhost:3000

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  التشغيل المباشر بدون Docker (Node.js)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

متطلبات: Node.js 18 أو أحدث

  RESOLVER_PASSWORD=كلمتك_هنا node src/server.js

على Windows:
  set RESOLVER_PASSWORD=كلمتك_هنا && node src/server.js

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  الإعدادات
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

RESOLVER_PASSWORD  كلمة المرور (إذا تركتها فارغة لا يوجد حماية)
PORT               المنفذ الافتراضي 3000
